214 private void OnTriggerEnter2D(Collider2D collision)
215 {
216 if (collision.gameObject.CompareTag("Door"))
217 {
218 Door tempDoor = collision.gameObject.GetComponent<Door>();
219 // Active doors require interaction whereas passive doors do not.
220 if (!tempDoor.IsActive())
221 {
222 if (lastInteractedDoor == null && justTraveled == false)
223 {
224 if (SceneManager.GetActiveScene().name == tempDoor.GetDestinationSceneName())
225 {
226 _isTeleporting = true;
227 transform.position = tempDoor.GetDestinationLocation();
228 Door[] doorObjects = FindObjectsByType<Door>(FindObjectsSortMode.None);
229 foreach (Door door in doorObjects)
230 {
231 if (door.transform.position == transform.position)
232 {
233 lastInteractedDoor = door;
234 break;
235 }
236 }
237 Invoke(nameof(UnlockTeleport), .5f);
238 }
239 else
240 {
241 _isTeleporting = true;
242 SceneManager.LoadScene(tempDoor.GetDestinationSceneName());
243 transform.position = tempDoor.GetDestinationLocation();
244 Door[] doorObjects = FindObjectsByType<Door>(FindObjectsSortMode.None);
245 foreach (Door door in doorObjects)
246 {
247 if (math.abs(door.transform.position.x - transform.position.x) < 1 && math.abs(door.transform.position.y - transform.position.y) < 1)
248 {
249 lastInteractedDoor = door;
250 break;
251 }
252 }
253 Invoke(nameof(UnlockTeleport), .5f);
254 }
255
256 justTraveled = true;
257 }
258 }
259 }
260 }

287 private void Interact()
288 {
289 var facingDir = new Vector3(characterAnimator.GetFloat("moveX"), characterAnimator.GetFloat("moveY"));
290 var interactPosition = transform.position + facingDir;
291 var collider = Physics2D.OverlapCircle(interactPosition, 0.2f, _interactableObjectsLayer);
292 if (collider != null)
293 {
294 if (collider.TryGetComponent<QuestHandler>(out QuestHandler qh))
295 {
296 int handledQuestId = ((IQuestOwner)this).GetHandledQuestId(qh);
297 if (handledQuestId != -1 && ((IQuestOwner)this).IsQuestCompleteable(handledQuestId))
298 {
299 QuestFramework.GetQuestFramework().CompleteQuest(handledQuestId, qh, (IQuestOwner)this);
300 }
301
302 else
303 {
304 collider.GetComponent<InteractableObject>()?.Interact();
305 }
306 }
307 else
308 {
309 collider.GetComponent<InteractableObject>()?.Interact();
310 }
311
312 }
