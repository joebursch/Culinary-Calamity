193 private void Interact()
194 {
195 var facingDir = new Vector3(characterAnimator.GetFloat("moveX"), characterAnimator.GetFloat("moveY"));
196 var interactPosition = transform.position + facingDir;
197 var collider = Physics2D.OverlapCircle(interactPosition, 0.2f, _interactableObjectsLayer);
198 if (collider != null) { collider.GetComponent<InteractableObject>()?.Interact(); }
199 }

205 void OnCollisionEnter2D(Collision2D collision)
206 {
207 if (collision.gameObject.layer == LayerMask.NameToLayer("Items"))
208 {
209 Item item = collision.gameObject.GetComponent<Item>();
210 _playerInventory.AddItem(item.GetItemId());
211 Destroy(collision.gameObject);
212 }
213 else if (collision.gameObject.layer == LayerMask.NameToLayer("InteractableObjects"))
214 {
215 if (collision.gameObject.CompareTag("Door"))
216 {
217 Door tempDoor = collision.gameObject.GetComponent<Door>();
218
219 if (!tempDoor.IsActive())
220 {
221 if (SceneManager.GetActiveScene().name == tempDoor.GetDestinationSceneName())
222 {
223 transform.position = tempDoor.GetDestinationLocation();
224 }
225 else
226 {
227 SceneManager.LoadScene(tempDoor.GetDestinationSceneName());
228 transform.position = tempDoor.GetDestinationLocation();
229 }
230 }
231 }
232 }
233 else if (collision.gameObject.layer == LayerMask.NameToLayer("Projectiles"))
234 {
235 TakeDamage(collision.gameObject.GetComponent<Projectile>().GetProjectileDamage());
236 }
237 }
